void maxim(v, stanga, dreapta, max)
	daca s == d
		daca v(stanga) > max
			max = v(stanga)
		return
	daca dreapta - stanga == 1 (mai sunt doar 2 elemente)
		daca v(stanga) > v(dreapta)
			daca v(stanga) > max
				max = v(stanga)
				return
		altfel
			dacav(dreapta) > max
				max = v(dreapta)
				return
	mijloc = (stanga + dreapta) / 2
	maxim(v, stanga, mijloc, max)
	maxim(v, mijloc + 1, dreapta, max)

void search(v, st, dr, x, freq)
	daca st == dr si v(st) == x
		freq++
		return
	mij = (st+dr)/2
	search(v, st, mij, x, freq)
	search(v, mij+1, dr, x, freq)
